00233 std::string uFormatv (const char *fmt, va_list args)
00234 {
00235     // Allocate a buffer on the stack that's big enough for us almost
00236     // all the time.  Be prepared to allocate dynamically if it doesn't fit.
00237     size_t size = 1024;
00238     std::vector<char> dynamicbuf(size);
00239     char *buf = &dynamicbuf[0];
00240 
00241     va_list argsTmp;
00242 
00243     while (1) {
00244         va_copy(argsTmp, args);
00245 
00246         // Try to vsnprintf into our buffer.
00247         int needed;
00248         if(argsTmp != 0)
00249         {
00250                 needed = vsnprintf (buf, size, fmt, argsTmp);
00251         }
00252         else
00253         {
00254                 needed = snprintf (buf, size, "%s", fmt);
00255         }
00256         va_end(argsTmp);
00257         // NB. C99 (which modern Linux and OS X follow) says vsnprintf
00258         // failure returns the length it would have needed.  But older
00259         // glibc and current Windows return -1 for failure, i.e., not
00260         // telling us how much was needed.
00261         if (needed < (int)size-1 && needed >= 0) {
00262             // It fit fine so we're done.
00263             return std::string (buf, (size_t) needed);
00264         }
00265 
00266         // vsnprintf reported that it wanted to write more characters
00267         // than we allotted.  So try again using a dynamic buffer.  This
00268         // doesn't happen very often if we chose our initial size well.
00269         size = needed>=0?needed+2:size*2;
00270         dynamicbuf.resize (size);
00271         buf = &dynamicbuf[0];
00272     }
